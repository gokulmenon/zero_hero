async function fetchGeminiFact(name, zeros) { const response = await fetch('/api/fact', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `Tell me a fun, easy-to-understand fact or analogy about the number "${name}" (10^${zeros}). If it's a huge number, compare it to atoms in the universe or something similar. Keep it very short (max 2 sentences) and suitable for a student.` }] }] }) }); const data = await response.json(); return data; }
